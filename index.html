Explora y copia métodos de uso frecuente en Python para facilitar tus desarrollos.

split_custom()

Divide una cadena en una lista de palabras.

def split_custom(texto, separador=" "):
    resultado = []
    palabra = ""
    for char in texto:
        if char == separador:
            if palabra:
                resultado.append(palabra)
                palabra = ""
        else:
            palabra += char
    if palabra:
        resultado.append(palabra)
    return resultado

# Ejemplo
texto = "Hola cómo estás?"
resultado = split_custom(texto)  # ['Hola', 'cómo', 'estás?']

strip_custom()

Elimina los espacios en blanco al inicio y al final de una cadena.

def strip_custom(texto):
    inicio, fin = 0, len(texto) - 1
    while inicio <= fin and texto[inicio] == " ":
        inicio += 1
    while fin >= inicio and texto[fin] == " ":
        fin -= 1
    return texto[inicio:fin+1]

# Ejemplo
texto = "  Hola mundo  "
resultado = strip_custom(texto)  # "Hola mundo"

upper_custom()

Convierte todos los caracteres de una cadena a mayúsculas.

def upper_custom(texto):
    resultado = ""
    for char in texto:
        if 'a' <= char <= 'z':
            resultado += chr(ord(char) - 32)
        else:
            resultado += char
    return resultado

# Ejemplo
texto = "hola"
resultado = upper_custom(texto)  # "HOLA"

lower_custom()

Convierte todos los caracteres de una cadena a minúsculas.

def lower_custom(texto):
    resultado = ""
    for char in texto:
        if 'A' <= char <= 'Z':
            resultado += chr(ord(char) + 32)
        else:
            resultado += char
    return resultado

# Ejemplo
texto = "HOLA"
resultado = lower_custom(texto)  # "hola"

ordenar_custom()

Ordena una lista de elementos en orden ascendente.

def ordenar_custom(lista):
    for i in range(len(lista)):
        for j in range(i + 1, len(lista)):
            if lista[i] > lista[j]:
                lista[i], lista[j] = lista[j], lista[i]
    return lista

# Ejemplo
numeros = [3, 1, 4, 1, 5]
resultado = ordenar_custom(numeros)  # [1, 1, 3, 4, 5]

Escribir archivo CSV

Escribe información de listas en un archivo CSV.

import csv

FILE = 'datos.csv'
paises = ['Argentina', 'Perú', 'Chile', 'Bolivia', 'Paraguay']
capitales = ['Buenos Aires', 'Lima', 'Santiago de Chile', 'La Paz', 'Asunción']

with open(FILE, 'w', newline='', encoding='utf-8') as archivo:
    escritor = csv.writer(archivo)
    for i in range(len(paises)):
        escritor.writerow([paises[i], capitales[i]])

Escribir archivo TXT - write()

Escribe contenido en un archivo de texto usando write() en modo 'a' (append).

from random import randint

def crear_fila_none(f):
    res = [None]*f
    return res

def llenar_lista(fila):
    for i in range(len(fila)):
        fila[i] = randint(1,10)

def escribir_archivo(fname, contenido):
    fd = open(fname, 'a')
    for elemento in contenido:
        fd.write(str(elemento)+ ' ')
    fd.write('\n')
    fd.close()

FILE = 'filas.txt'
fila = crear_fila_none(5)
llenar_lista(fila)
escribir_archivo(FILE, fila)
print(fila)

Leer archivo TXT - read()

Lee todo el contenido de un archivo de texto de una vez usando read().

def leer_archivo(fn):
    fd = open(fn, 'r')
    cont = fd.read()
    fd.close()
    return cont

FILE = 'datos.txt'
contenido = leer_archivo(FILE)
print('CONTENIDO DEL ARCHIVO')
print(contenido)

Leer archivo TXT - readline()

Lee un archivo línea por línea usando readline().

def imprimir_archivo(fn):
    fd = open(fn, 'r')
    line = fd.readline()  # La primer línea la descarto
    while line:
        print(line)
        line = fd.readline()
    fd.close()
    return

FILE = 'datos.txt'
arch = imprimir_archivo(FILE)

Leer archivo TXT - for con fd

Lee un archivo iterando directamente sobre el descriptor de archivo.

def ver_for_fd(fn):
    fd = open(fn, 'r')
    res = []
    fd.readline()  # Leo e ignoro la primer línea
    for line in fd:
        print(line)
        res.append(line[:-1])
    fd.close()
    return res

FILE = 'datos.txt'
lista = ver_for_fd(FILE)
print(lista)
